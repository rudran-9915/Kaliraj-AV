class Solution {
    static int isPerfectNumber(long N) {
        if(N<=1){
            return 0;
        }
        long res =0;
        long sqrtN = (long)Math.sqrt(N);
        for(long i=1;i<=sqrtN;i++){
            if(N%i==0){
                res+=i;
                if(i!=1 && i!=N/i){
                    res+=N/i;
                }
            }
        }
        if(N==res){
            return 1;
        }
        return 0;
    }
}
